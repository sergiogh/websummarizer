var wpcom_remote_login_extra_auth = '';
function wpcom_remote_login_remove_dom_node_id( element_id ) {
	var dom_node = document.getElementById( element_id );
	if ( dom_node ) { dom_node.parentNode.removeChild( dom_node ); }
}
function wpcom_remote_login_remove_dom_node_classes( class_name ) {
	var dom_nodes = document.querySelectorAll( '.' + class_name );
	for ( var i = 0; i < dom_nodes.length; i++ ) {
		dom_nodes[ i ].parentNode.removeChild( dom_nodes[ i ] );
	}
}
function wpcom_remote_login_final_cleanup() {
	wpcom_remote_login_remove_dom_node_classes( "wpcom_remote_login_msg" );
	wpcom_remote_login_remove_dom_node_id( "wpcom_remote_login_key" );
	wpcom_remote_login_remove_dom_node_id( "wpcom_remote_login_validate" );
	wpcom_remote_login_remove_dom_node_id( "wpcom_remote_login_js" );
	wpcom_remote_login_remove_dom_node_id( "wpcom_request_access_iframe" );
	wpcom_remote_login_remove_dom_node_id( "wpcom_request_access_styles" );
}

// Watch for messages back from the remote login
window.addEventListener( "message", function( e ) {
	if ( e.origin === "https://r-login.wordpress.com" ) {
		var data = {};
		try {
			data = JSON.parse( e.data );
		} catch( e ) {
			wpcom_remote_login_final_cleanup();
			return;
		}

		if ( data.msg === 'LOGIN' ) {
			// Clean up the login check iframe
			wpcom_remote_login_remove_dom_node_id( "wpcom_remote_login_key" );

			var id_regex = new RegExp( /^[0-9]+$/ );
			var token_regex = new RegExp( /^.*|.*|.*$/ );
			if (
				token_regex.test( data.token )
				&& id_regex.test( data.wpcomid )
			) {
				// We have everything we need to ask for a login
				var script = document.createElement( "script" );
				script.setAttribute( "id", "wpcom_remote_login_validate" );
				script.src = '/remote-login.php?wpcom_remote_login=validate'
					+ '&wpcomid=' + data.wpcomid
					+ '&token=' + encodeURIComponent( data.token )
					+ '&host=' + window.location.protocol
					+ '//' + window.location.hostname
					+ '&postid=17836'
					+ '&is_singular=1';
				document.body.appendChild( script );
			}

			return;
		}

		// Safari ITP, not logged in, so redirect
		if ( data.msg === 'LOGIN-REDIRECT' ) {
			window.location = 'https://wordpress.com/log-in?redirect_to=' + window.location.href;
			return;
		}

		// Safari ITP, storage access failed, remove the request
		if ( data.msg === 'LOGIN-REMOVE' ) {
			var css_zap = 'html { -webkit-transition: margin-top 1s; transition: margin-top 1s; } /* 9001 */ html { margin-top: 0 !important; } * html body { margin-top: 0 !important; } @media screen and ( max-width: 782px ) { html { margin-top: 0 !important; } * html body { margin-top: 0 !important; } }';
			var style_zap = document.createElement( 'style' );
			style_zap.type = 'text/css';
			style_zap.appendChild( document.createTextNode( css_zap ) );
			document.body.appendChild( style_zap );

			var e = document.getElementById( 'wpcom_request_access_iframe' );
			e.parentNode.removeChild( e );

			document.cookie = 'wordpress_com_login_access=denied; path=/; max-age=31536000';

			return;
		}

		// Safari ITP
		if ( data.msg === 'REQUEST_ACCESS' ) {
			console.log( 'request access: safari' );

			// Check ITP iframe enable/disable knob
			if ( wpcom_remote_login_extra_auth !== 'safari_itp_iframe' ) {
				return;
			}

			// If we are in a "private window" there is no ITP.
			var private_window = false;
			try {
				var opendb = window.openDatabase( null, null, null, null );
			} catch( e ) {
				private_window = true;
			}

			if ( private_window ) {
				console.log( 'private window' );
				return;
			}

			var iframe = document.createElement( 'iframe' );
			iframe.id = 'wpcom_request_access_iframe';
			iframe.setAttribute( 'scrolling', 'no' );
			iframe.setAttribute( 'sandbox', 'allow-storage-access-by-user-activation allow-scripts allow-same-origin allow-top-navigation-by-user-activation' );
			iframe.src = 'https://r-login.wordpress.com/remote-login.php?wpcom_remote_login=request_access&origin=' + encodeURIComponent( data.origin ) + '&wpcomid=' + encodeURIComponent( data.wpcomid );

			var css = 'html { -webkit-transition: margin-top 1s; transition: margin-top 1s; } /* 9001 */ html { margin-top: 46px !important; } * html body { margin-top: 46px !important; } @media screen and ( max-width: 660px ) { html { margin-top: 71px !important; } * html body { margin-top: 71px !important; } #wpcom_request_access_iframe { display: block; height: 71px !important; } } #wpcom_request_access_iframe { border: 0px; height: 46px; position: fixed; top: 0; left: 0; width: 100%; min-width: 100%; z-index: 99999; background: #23282d; } ';

			var style = document.createElement( 'style' );
			style.type = 'text/css';
			style.id = 'wpcom_request_access_styles';
			style.appendChild( document.createTextNode( css ) );
			document.body.appendChild( style );

			document.body.appendChild( iframe );
		}

		if ( data.msg === 'DONE' ) {
			wpcom_remote_login_final_cleanup();
		}
	}
}, false );

}, false );

	<header class="entry-header">
		<h1 class="entry-title">What is next in quantum&nbsp;advantage?</h1>

				<div class="entry-meta">
						<span class="sep">Posted on </span><a href="https://quantumfrontiers.com/2026/02/28/what-is-next-in-quantum-advantage/" title="12:01 pm" rel="bookmark"><time class="entry-date" datetime="2026-02-28T12:01:25-08:00">February 28, 2026</time></a><span class="by-author"> <span class="sep"> by </span> <span class="author vcard"><a class="url fn n" href="https://quantumfrontiers.com/author/mail3373492c578/" title="View all posts by Dominik Hangleiter" rel="author">Dominik Hangleiter</a></span></span>		</div><!-- .entry-meta -->
			</header><!-- .entry-header -->

	<div class="entry-content">
		
<p class="wp-block-paragraph">We are now at an exciting point in our process of developing quantum computers and understanding their computational power: It has been demonstrated that quantum computers <em>can</em> outperform classical ones (if you buy my argument from <a href="https://quantumfrontiers.com/2026/01/06/has-quantum-advantage-been-achieved/">Parts 1</a> and <a href="https://quantumfrontiers.com/2026/01/25/has-quantum-advantage-been-achieved-part-2-considering-the-evidence/">2</a> of this mini series). And it has been demonstrated that quantum fault-tolerance is possible for at least a few logical qubits. Together, these form the elementary building blocks of useful quantum computing.</p>



<p class="wp-block-paragraph">And yet: the devices we have seen so far are still nowhere near being useful for any advantageous application in, say, condensed-matter physics or quantum chemistry, which is where the promise of quantum computers lies.</p>



<p class="wp-block-paragraph">So what is next in quantum advantage?</p>



<p class="wp-block-paragraph">This is what this third and last part of my mini-series on the question &#8220;Has quantum advantage been achieved?&#8221; is about.</p>



<blockquote class="wp-block-quote is-layout-flow wp-block-quote-is-layout-flow">
<p class="wp-block-paragraph"><strong><em>The 100 logical qubits regime</em></strong><br>I want to have in mind the regime in which we have 100 well-functioning logical qubits, so 100 qubits on which we can run maybe 100 000 gates.</p>
</blockquote>



<p class="wp-block-paragraph">Building devices operating in this regime will require thousand(s) of physical qubits and is therefore well beyond the proof-of-principle quantum advantage and fault-tolerance experiments that have been done. At the same time, it is (so far) still one or more orders of magnitude away from any of the first applications such as simulating, say, the Fermi-Hubbard model or breaking cryptography. In other words, it is a qualitatively different regime from the early fault-tolerant computations we can do now. And yet, there is not a clear picture for what we can and should do with such devices.</p>



<h2 class="wp-block-heading">The next milestone: classically verifiable quantum advantage</h2>



<p class="wp-block-paragraph">In this post, I want to argue that a key milestone we should aim for in the 100 logical qubit regime is <em>classically verifiable quantum advantage</em>. Achieving this will not only require the jump in quantum device capabilities but also finding advantage schemes that allow for classical verification using these limited resources.</p>



<p class="wp-block-paragraph">Why is it an interesting and feasible goal and what is it anyway?</p>



<p class="wp-block-paragraph">To my mind, the biggest weakness of the RCS experiments is the way they are verified. I discussed this extensively in the last posts&#8212;verification uses XEB which can be classically spoofed, and only actually measured in the simulatable regime. Really, in a quantum advantage experiment I would want there to be an efficient procedure that will without any reasonable doubt convince us that a computation <em>must have</em> been performed by a quantum computer when we run it. In what I think of as classically verifiable quantum advantage, a (classical) verifier would come up with challenge circuits which they would then send to a quantum server. These would be designed in such a way that once the server returns <em>classical</em> samples from those circuits, the verifier can convince herself that the server must have run a quantum computation.</p>



<figure class="wp-block-image size-large"><img data-attachment-id="17838" data-permalink="https://quantumfrontiers.com/2026/02/28/what-is-next-in-quantum-advantage/verifier-prover/" data-orig-file="https://quantumfrontiers.com/wp-content/uploads/2026/02/verifier-prover.png" data-orig-size="2816,738" data-comments-opened="1" data-image-meta="{&quot;aperture&quot;:&quot;0&quot;,&quot;credit&quot;:&quot;&quot;,&quot;camera&quot;:&quot;&quot;,&quot;caption&quot;:&quot;&quot;,&quot;created_timestamp&quot;:&quot;0&quot;,&quot;copyright&quot;:&quot;&quot;,&quot;focal_length&quot;:&quot;0&quot;,&quot;iso&quot;:&quot;0&quot;,&quot;shutter_speed&quot;:&quot;0&quot;,&quot;title&quot;:&quot;&quot;,&quot;orientation&quot;:&quot;0&quot;}" data-image-title="verifier-prover" data-image-description="" data-image-caption="" data-medium-file="https://quantumfrontiers.com/wp-content/uploads/2026/02/verifier-prover.png?w=300" data-large-file="https://quantumfrontiers.com/wp-content/uploads/2026/02/verifier-prover.png?w=584" width="1024" height="268" src="https://quantumfrontiers.com/wp-content/uploads/2026/02/verifier-prover.png?w=1024" alt="" class="wp-image-17838" srcset="https://quantumfrontiers.com/wp-content/uploads/2026/02/verifier-prover.png?w=1024 1024w, https://quantumfrontiers.com/wp-content/uploads/2026/02/verifier-prover.png?w=2048 2048w, https://quantumfrontiers.com/wp-content/uploads/2026/02/verifier-prover.png?w=150 150w, https://quantumfrontiers.com/wp-content/uploads/2026/02/verifier-prover.png?w=300 300w, https://quantumfrontiers.com/wp-content/uploads/2026/02/verifier-prover.png?w=768 768w, https://quantumfrontiers.com/wp-content/uploads/2026/02/verifier-prover.png?w=1440 1440w" sizes="(max-width: 1024px) 100vw, 1024px" /><figcaption class="wp-element-caption"><em>The theoretical computer scientist&#8217;s cartoon of verifying a quantum computer.</em></figcaption></figure>



<p class="wp-block-paragraph">This is the jump from a physics-type experiment (the sense in which advantage has been achieved) to a secure protocol that can be used in settings where I do not want to trust the server and the data it provides me with. Such security may also allow a first application of quantum computers: to generate random numbers whose genuine randomness can be certified&#8212;a task that is impossible classically.</p>



<p class="wp-block-paragraph">Here is the problem: On the one hand, we do know of schemes that allow us to classically verify that a computer is quantum and generate random numbers, so called <em>cryptographic proofs of quantumness</em> (<a href="https://doi.org/10.1109/FOCS.2018.00038">PoQ</a>). A proof of quantumness is a highly reliable scheme in that its security relies on well-established cryptography. Their big drawback is that they require a large number of qubits and operations, comparable to the resources required for factoring. On the other hand, the computations we <em>can</em> run in the advantage regime&#8212;basically, random circuits&#8212;are very resource-efficient but not verifiable.</p>



<p class="wp-block-paragraph">The 100-logical-qubit regime lies right in the middle, and it seems more than plausible that classically verifiable advantage is possible in this regime. The theory challenge ahead of us is to find it: a quantum advantage scheme that is very resource-efficient like RCS and also classically verifiable like proofs of quantumness.</p>

<p class="wp-block-paragraph">With this in mind, let me spell out some concrete goals that we can achieve using 100 logical qubits on the road to classically verifiable quantum advantage.</p>



<h2 class="wp-block-heading">1. Demonstrate fault-tolerant quantum advantage</h2>



<p class="wp-block-paragraph">Before we talk about verifiable advantage, the first experiment I would like to see is one that combines the two big achievements of the past years, and shows that quantum advantage and fault-tolerance can be achieved simultaneously. Such an experiment would be similar in type to the RCS experiments, but run on encoded qubits with gate sets that match that encoding. During the computation, noise would be suppressed by correcting for errors using the code. In doing so, we could reach the near-perfect regime of RCS as opposed to the finite-fidelity regime that current RCS experiments operate in (as I discussed in detail in <a href="https://quantumfrontiers.com/2026/01/25/has-quantum-advantage-been-achieved-part-2-considering-the-evidence/">Part 2</a>).</p>



<p class="wp-block-paragraph">Random circuits with a quantum advantage that are particularly easy to implement fault-tolerantly are so-called <em>IQP circuits</em>. In those circuits, the gates are controlled-NOT gates and diagonal gates, so rotations <math data-latex="Z(\theta,z)"><semantics><mrow><mi>Z</mi><mo form="prefix" stretchy="false">(</mo><mi>θ</mi><mo separator="true">,</mo><mi>z</mi><mo form="postfix" stretchy="false">)</mo></mrow><annotation encoding="application/x-tex">Z(\theta,z)</annotation></semantics></math>, which just add a phase to a basis state <math data-latex="\ket x"><semantics><mpadded><mi>|</mi><mi>x</mi><mo form="postfix" stretchy="false">⟩</mo></mpadded><annotation encoding="application/x-tex">\ket x</annotation></semantics></math> as <math data-latex="Z(\theta, z) \ket x = \exp(i \theta \, z \cdot x) \ket x"><semantics><mrow><mi>Z</mi><mo form="prefix" stretchy="false">(</mo><mi>θ</mi><mo separator="true">,</mo><mi>z</mi><mo form="postfix" stretchy="false">)</mo><mpadded><mi>|</mi><mi>x</mi><mo form="postfix" stretchy="false">⟩</mo></mpadded><mo>=</mo><mrow><mi>exp</mi><mo>⁡</mo></mrow><mo form="prefix" stretchy="false">(</mo><mi>i</mi><mi>θ</mi><mspace width="0.1667em"></mspace><mi>z</mi><mo>⋅</mo><mi>x</mi><mo form="postfix" stretchy="false">)</mo><mpadded><mi>|</mi><mi>x</mi><mo form="postfix" stretchy="false">⟩</mo></mpadded></mrow><annotation encoding="application/x-tex">Z(\theta, z) \ket x = \exp(i \theta \, z \cdot x) \ket x</annotation></semantics></math>. The only &#8220;quantumness&#8221; comes from the fact that each input qubit is in the superposition state <math data-latex="\ket + = \ket 0 + \ket 1"><semantics><mrow><mpadded><mi>|</mi><mo lspace="0em" rspace="0em">+</mo><mo form="postfix" stretchy="false">⟩</mo></mpadded><mo>=</mo><mpadded><mi>|</mi><mn>0</mn><mo form="postfix" stretchy="false">⟩</mo></mpadded><mo>+</mo><mpadded><mi>|</mi><mn>1</mn><mo form="postfix" stretchy="false">⟩</mo></mpadded></mrow><annotation encoding="application/x-tex">\ket + = \ket 0 + \ket 1</annotation></semantics></math>, and that all qubits are measured in the <math data-latex="X"><semantics><mi>X</mi><annotation encoding="application/x-tex">X</annotation></semantics></math> basis. This is an example of an example of an IQP circuit:</p>

<p class="wp-block-paragraph">As it so happens, IQP circuits are already really well understood since one of the first proposals for quantum advantage was based on IQP circuits (<a href="http://rspa.royalsocietypublishing.org/content/465/2105/1413">VerIQP1</a>), and for a lot of the results in random circuits, we have precursors for IQP circuits, in particular, their ideal and noisy complexity (<a href="https://quantum-journal.org/papers/q-2017-04-25-8/">SimIQP</a>). This is because their near-classical structure makes them relatively easy to study. Most importantly, their outcome probabilities are simple (but exponentially large) sums over phases <math data-latex="e^{i \theta }"><semantics><msup><mi>e</mi><mrow><mi>i</mi><mi>θ</mi></mrow></msup><annotation encoding="application/x-tex">e^{i \theta }</annotation></semantics></math> that can just be read off from which gates are applied in the circuit and we can use well-established classical techniques like Boolean analysis and coding theory to understand those.</p>



<p class="wp-block-paragraph">IQP gates are natural for fault-tolerance because there are codes in which all the operations involved can be implemented <em>transversally</em>. This means that they only require parallel physical single- or two-qubit gates to implement a logical gate rather than complicated fault-tolerant protocols which are required for universal circuits. This is in stark contrast to universal circuit which require resource-intensive fault-tolerant protocols. Running computations with IQP circuits would also be a step towards running real computations in that they can involve structured components such as cascades of CNOT gates and the like. These show up all over fault-tolerant constructions of algorithmic primitives such as arithmetic or phase estimation circuits.</p>



<p class="wp-block-paragraph">Our concrete proposal for an IQP-based fault-tolerant quantum advantage experiment in reconfigurable-atom arrays is based on interleaving diagonal gates and CNOT gates to achieve super-fast scrambling (<a href="https://link.aps.org/doi/10.1103/PRXQuantum.6.020338">ftIQP1</a>). A medium-size version of this protocol was implemented by the Harvard group (<a href="https://www.nature.com/articles/s41586-023-06927-3">LogicalExp</a>) but with only a bit more effort, it could be performed in the advantage regime.</p>



<p class="wp-block-paragraph">In those proposals, verification will still suffer from the same problems of standard RCS experiments, so what&#8217;s up next is to fix that!</p>



<h2 class="wp-block-heading">2. Closing the verification loophole</h2>



<p class="wp-block-paragraph">I said that a key milestone for the 100-logical-qubit regime is to find schemes that lie in between RCS and proofs of quantumness in terms of their resource requirements but at the same time allow for more efficient and more convincing verification than RCS. Naturally, there are two ways to approach this space&#8212;we can make quantum advantage schemes more verifiable, and we can make proofs of quantumness more resource-efficient.</p>



<p class="wp-block-paragraph">First, let&#8217;s focus on the former approach and set a more moderate goal than full-on classical verification of data from an untrusted server. Are there variants of RCS that allow us to efficiently verify that <em>finite-fidelity RCS</em> has been achieved if we trust the experimenter and the data they hand us?</p>



<h4 class="wp-block-heading"><strong>2.1 Efficient quantum verification using random circuits with symmetries</strong></h4>



<p class="wp-block-paragraph">Indeed, there are! I like to think of the schemes that achieve this as <em>random circuits with symmetries</em>. A symmetry is an operator <math data-latex="S "><semantics><mi>S</mi><annotation encoding="application/x-tex">S </annotation></semantics></math> such that the outcome state of the computation <math data-latex="\ket C"><semantics><mpadded><mi>|</mi><mi>C</mi><mo form="postfix" stretchy="false">⟩</mo></mpadded><annotation encoding="application/x-tex">\ket C</annotation></semantics></math> (or some intermediate state) is invariant under the symmetry, so <math data-latex="S \ket C =\ket C"><semantics><mrow><mi>S</mi><mpadded><mi>|</mi><mi>C</mi><mo form="postfix" stretchy="false">⟩</mo></mpadded><mo>=</mo><mpadded><mi>|</mi><mi>C</mi><mo form="postfix" stretchy="false">⟩</mo></mpadded></mrow><annotation encoding="application/x-tex">S \ket C =\ket C</annotation></semantics></math>. The idea is then to find circuits <math data-latex="C"><semantics><mi>C</mi><annotation encoding="application/x-tex">C</annotation></semantics></math> that exhibit a quantum advantage and at the same time have symmetries that can be easily measured, say, using only single-qubit measurements or a single gate layer. Then, we can use these measurements to check whether or not the pre-measurement state respects the symmetries. This is a test for whether the quantum computer prepared the correct state, because errors or deviations from the true state would violate the symmetry (unless they were adversarially engineered). </p>



<p class="wp-block-paragraph">In random circuits with symmetries, we can thus use small, well-characterized measurements whose outcomes we trust to probe whether a large quantum circuit has been run correctly. This is possible in a scenario I call the <em>trusted experimenter scenario</em>.</p>



<blockquote class="wp-block-quote is-layout-flow wp-block-quote-is-layout-flow">
<p class="wp-block-paragraph"><strong><em>The trusted experimenter scenario</em></strong><br>In this scenario, we receive data from an actual experiment in which we trust that certain measurements were actually and correctly performed.</p>
</blockquote>


<div class="wp-block-image">
<figure class="aligncenter size-large is-resized"><img data-attachment-id="17844" data-permalink="https://quantumfrontiers.com/2026/02/28/what-is-next-in-quantum-advantage/circuit-measurements/" data-orig-file="https://quantumfrontiers.com/wp-content/uploads/2026/02/circuit-measurements.png" data-orig-size="1440,1196" data-comments-opened="1" data-image-meta="{&quot;aperture&quot;:&quot;0&quot;,&quot;credit&quot;:&quot;&quot;,&quot;camera&quot;:&quot;&quot;,&quot;caption&quot;:&quot;&quot;,&quot;created_timestamp&quot;:&quot;0&quot;,&quot;copyright&quot;:&quot;&quot;,&quot;focal_length&quot;:&quot;0&quot;,&quot;iso&quot;:&quot;0&quot;,&quot;shutter_speed&quot;:&quot;0&quot;,&quot;title&quot;:&quot;&quot;,&quot;orientation&quot;:&quot;0&quot;}" data-image-title="circuit-measurements" data-image-description="" data-image-caption="" data-medium-file="https://quantumfrontiers.com/wp-content/uploads/2026/02/circuit-measurements.png?w=300" data-large-file="https://quantumfrontiers.com/wp-content/uploads/2026/02/circuit-measurements.png?w=584" loading="lazy" width="1024" height="850" src="https://quantumfrontiers.com/wp-content/uploads/2026/02/circuit-measurements.png?w=1024" alt="" class="wp-image-17844" style="aspect-ratio:1.1979708827050832;width:412px;height:auto" srcset="https://quantumfrontiers.com/wp-content/uploads/2026/02/circuit-measurements.png?w=1024 1024w, https://quantumfrontiers.com/wp-content/uploads/2026/02/circuit-measurements.png?w=150 150w, https://quantumfrontiers.com/wp-content/uploads/2026/02/circuit-measurements.png?w=300 300w, https://quantumfrontiers.com/wp-content/uploads/2026/02/circuit-measurements.png?w=768 768w, https://quantumfrontiers.com/wp-content/uploads/2026/02/circuit-measurements.png 1440w" sizes="(max-width: 1024px) 100vw, 1024px" /><figcaption class="wp-element-caption"><em>I think of random circuits with symmetries as introducing measurements in the circuit that check for errors.</em></figcaption></figure>
</div>


<p class="wp-block-paragraph">Here are some examples of random circuits with symmetries, which allow for efficient verification of quantum advantage in the trusted experimenter scenario.</p>



<p class="wp-block-paragraph"><strong>Graph states.</strong> My first example are locally rotated graph states (<a href="https://www.nature.com/articles/s41467-024-55342-3">GStates</a>). These are states that are prepared by CZ gates acting according to the edges of a graph on an initial all-<math data-latex="\ket +"><semantics><mpadded><mi>|</mi><mo lspace="0em" rspace="0em">+</mo><mo form="postfix" stretchy="false">⟩</mo></mpadded><annotation encoding="application/x-tex">\ket +</annotation></semantics></math> state, and a layer of single-qubit <math data-latex="Z"><semantics><mi>Z</mi><annotation encoding="application/x-tex">Z</annotation></semantics></math>-rotations is performed before a measurement in the <math data-latex="X"><semantics><mi>X</mi><annotation encoding="application/x-tex">X</annotation></semantics></math> basis. (Yes, this is also an IQP circuit.) The symmetries of this circuit are locally rotated Pauli operators, and can therefore be measured using only single-qubit rotations and measurements. What is more, these symmetries fully determine the graph state. Determining the fidelity then just amounts to averaging the expectation values of the symmetries, which is so efficient you can even do it in your head. In this example, we need measuring the outcome state to obtain hard-to-reproduce samples and measuring the symmetries are done in two different (single-qubit) bases.</p>



<p class="wp-block-paragraph">With 100 logical qubits, samples from classically intractable graph states on several 100 qubits could be easily generated.</p>



<p class="wp-block-paragraph"><strong>Bell sampling.</strong> The drawback of this approach is that we need to make two different measurements for verification and sampling. But it would be much more neat if we could just verify the correctness of a set of classically hard samples by <em>only using those samples</em>. For an example where this is possible, consider two copies of the output state <math data-latex="\ket C"><semantics><mpadded><mi>|</mi><mi>C</mi><mo form="postfix" stretchy="false">⟩</mo></mpadded><annotation encoding="application/x-tex">\ket C</annotation></semantics></math> of a random circuit, so <math data-latex="\ket C \otimes \ket C"><semantics><mrow><mpadded><mi>|</mi><mi>C</mi><mo form="postfix" stretchy="false">⟩</mo></mpadded><mo>⊗</mo><mpadded><mi>|</mi><mi>C</mi><mo form="postfix" stretchy="false">⟩</mo></mpadded></mrow><annotation encoding="application/x-tex">\ket C \otimes \ket C</annotation></semantics></math>. This state is invariant under a swap of the two copies, and in fact the expectation value of the SWAP operator <math data-latex="\mathbb S"><semantics><mi>𝕊</mi><annotation encoding="application/x-tex">\mathbb S</annotation></semantics></math> in a noisy state preparation <math data-latex="\rho_C"><semantics><msub><mi>ρ</mi><mi>C</mi></msub><annotation encoding="application/x-tex">\rho_C</annotation></semantics></math> of <math data-latex="\ket C \otimes \ket C"><semantics><mrow><mpadded><mi>|</mi><mi>C</mi><mo form="postfix" stretchy="false">⟩</mo></mpadded><mo>⊗</mo><mpadded><mi>|</mi><mi>C</mi><mo form="postfix" stretchy="false">⟩</mo></mpadded></mrow><annotation encoding="application/x-tex">\ket C \otimes \ket C</annotation></semantics></math> determines the purity of the state, so <math data-latex="P = \mathrm{tr}[ \rho \mathbb S]"><semantics><mrow><mi>P</mi><mo>=</mo><mrow><mtext></mtext><mi>tr</mi></mrow><mo form="prefix" stretchy="false">[</mo><mi>ρ</mi><mi>𝕊</mi><mo form="postfix" stretchy="false">]</mo></mrow><annotation encoding="application/x-tex">P = \mathrm{tr}[ \rho \mathbb S]</annotation></semantics></math>. It turns out that measuring all pairs of qubits in the state <math data-latex="\ket C \otimes \ket C"><semantics><mrow><mpadded><mi>|</mi><mi>C</mi><mo form="postfix" stretchy="false">⟩</mo></mpadded><mo>⊗</mo><mpadded><mi>|</mi><mi>C</mi><mo form="postfix" stretchy="false">⟩</mo></mpadded></mrow><annotation encoding="application/x-tex">\ket C \otimes \ket C</annotation></semantics></math> in the pairwise basis of the four Bell states <math data-latex="\ket{\sigma} = (\sigma \otimes \mathbb 1) (\ket{00} + \ket{11})"><semantics><mrow><mpadded><mi>|</mi><mi>σ</mi><mo form="postfix" stretchy="false">⟩</mo></mpadded><mo>=</mo><mo form="prefix" stretchy="false">(</mo><mi>σ</mi><mo>⊗</mo><mn>𝟙</mn><mo form="postfix" stretchy="false">)</mo><mo form="prefix" stretchy="false">(</mo><mpadded><mi>|</mi><mn>00</mn><mo form="postfix" stretchy="false">⟩</mo></mpadded><mo>+</mo><mpadded><mi>|</mi><mn>11</mn><mo form="postfix" stretchy="false">⟩</mo></mpadded><mo form="postfix" stretchy="false">)</mo></mrow><annotation encoding="application/x-tex">\ket{\sigma} = (\sigma \otimes \mathbb 1) (\ket{00} + \ket{11})</annotation></semantics></math>, where <math data-latex="\sigma"><semantics><mi>σ</mi><annotation encoding="application/x-tex">\sigma</annotation></semantics></math> is one of the four Pauli matrices <math data-latex="\mathbb 1, X, Y, Z"><semantics><mrow><mn>𝟙</mn><mo separator="true">,</mo><mi>X</mi><mo separator="true">,</mo><mi>Y</mi><mo separator="true">,</mo><mi>Z</mi></mrow><annotation encoding="application/x-tex">\mathbb 1, X, Y, Z</annotation></semantics></math>, this is hard to simulate classically (<a href="https://link.aps.org/doi/10.1103/PhysRevLett.133.020601">BellSamp</a>). You may also observe that the SWAP operator is diagonal in the Bell basis, so its expectation value can be extracted from the Bell-basis measurements&#8212;our hard to simulate samples. To do this, we just average sign assignments to the samples according to their parity.</p>



<p class="wp-block-paragraph">If the circuit is random, then under the same assumptions as those used in XEB for random circuits, the purity is a good estimator of the fidelity, so <math data-latex="P \approx F(\rho_C, \ket C \otimes \ket C)"><semantics><mrow><mi>P</mi><mo>≈</mo><mi>F</mi><mo form="prefix" stretchy="false">(</mo><msub><mi>ρ</mi><mi>C</mi></msub><mo separator="true">,</mo><mpadded><mi>|</mi><mi>C</mi><mo form="postfix" stretchy="false">⟩</mo></mpadded><mo>⊗</mo><mpadded><mi>|</mi><mi>C</mi><mo form="postfix" stretchy="false">⟩</mo></mpadded><mo form="postfix" stretchy="false">)</mo></mrow><annotation encoding="application/x-tex">P \approx F(\rho_C, \ket C \otimes \ket C)</annotation></semantics></math>. So here is an example, where <em>efficient</em> verification is possible directly from hard-to-simulate classical samples under the same assumptions as those used to argue that XEB equals fidelity.</p>



<p class="wp-block-paragraph">With 100 logical qubits, we can achieve quantum advantage which is at least as hard as the current RCS experiments that can also be efficiently (physics-)verified from the classical data.</p>



<p class="wp-block-paragraph"><strong>Fault-tolerant circuits.</strong> Finally, suppose that we run a fault-tolerant quantum advantage experiment. Then, there is a natural set of symmetries of the state at any point in the circuit, namely, the stabilizers of the code we use. In a fault-tolerant experiment we repeatedly measure those stabilizers mid-circuit, so why not use that data to assess the quality of the logical state? Indeed, it turns out that the logical fidelity can be estimated efficiently from stabilizer expectation values even in situations in which the logical circuit has a quantum advantage (<a href="https://doi.org/10.48550/arXiv.2601.21472">SyndFid</a>).</p>



<p class="wp-block-paragraph">With 100 logical qubits, we could therefore just run fault-tolerant IQP circuits in the advantage regime (<a href="https://link.aps.org/doi/10.1103/PRXQuantum.6.020338">ftIQP1</a>) and the syndrome data would allow us to estimate the logical fidelity.</p>



<hr class="wp-block-separator has-alpha-channel-opacity" />



<p class="wp-block-paragraph">In all of these examples of random circuits with symmetries, coming up with classical samples that pass the verification tests is very easy, so the trusted-experimenter scenario is crucial for this to work. (Note, however, that it may be possible to <a href="https://www.youtube.com/watch?v=Cnm7XSz5VQM">add tests to Bell sampling</a> that make spoofing difficult.) At the same time, these proposals are very resource-efficient in that they only increase the cost of a pure random-circuit experiment by a relatively small amount. What is more, the required circuits have more structure than random circuits in that they typically require gates that are natural in fault-tolerant implementations of quantum algorithms.</p>



<p class="wp-block-paragraph">Performing random circuit sampling with symmetries is therefore a natural next step en-route to both classically verifiable advantage that closes the <em>no-efficient verification</em> loophole, and towards implementing actual algorithms.</p>



<p class="wp-block-paragraph">What if we do not want to afford that level of trust in the person who runs the quantum circuit, however?</p>



<h4 class="wp-block-heading"><strong>2.2 Classical verification using random circuits with planted secrets</strong></h4>



<p class="wp-block-paragraph">If we do not trust the experimenter, we are in the <em>untrusted quantum server scenario</em>.</p>



<blockquote class="wp-block-quote is-layout-flow wp-block-quote-is-layout-flow">
<p class="wp-block-paragraph"><strong><em>The untrusted quantum server scenario</em></strong><br>In this scenario, we delegate a quantum computation to an untrusted (presumably remote) quantum server&#8212;think of using a Google or Amazon cloud server to run your computation. We can communicate with this server using classical information.</p>
</blockquote>



<p class="wp-block-paragraph">In the untrusted server scenario, we can hope to use ideas from proofs of quantumness such as the use of classical cryptography to design families of quantum circuits in which some secret structure is planted. This secret structure should give the verifier a way to check whether a set of samples passes a certain verification test. At the same time it should not be detectable, or at least not be identifiable from the circuit description alone.</p>



<p class="wp-block-paragraph">The simplest example of such secret structure could be a large peak in an otherwise flat output distribution of a random-looking quantum circuit. To do this, the verifier would pick a (random) string <math data-latex="x"><semantics><mi>x</mi><annotation encoding="application/x-tex">x</annotation></semantics></math> and design a circuit such that the probability of seeing <math data-latex="x"><semantics><mi>x</mi><annotation encoding="application/x-tex">x</annotation></semantics></math> in samples, is large. If the peak is hidden well, finding it just from the circuit description would require searching through all of the <math data-latex="2^n"><semantics><msup><mn>2</mn><mi>n</mi></msup><annotation encoding="application/x-tex">2^n</annotation></semantics></math> outcome bit strings and even just determining one of the outcome probabilities is exponentially difficult. A classical spoofer trying to fake the samples from a quantum computer would then be caught immediately: the list of samples they hand the verifier will not even contain <math data-latex="x"><semantics><mi>x</mi><annotation encoding="application/x-tex">x</annotation></semantics></math> unless they are unbelievably lucky, since there are exponentially many possible choices of <math data-latex="x"><semantics><mi>x</mi><annotation encoding="application/x-tex">x</annotation></semantics></math>.</p>



<p class="wp-block-paragraph">Unfortunately, planting such secrets seems to be very difficult using universal circuits, since the output distributions are so unstructured. This is why we have not yet found good candidates of circuits with peaks, but some tries have been made (<a href="http://arxiv.org/abs/2404.14493">Peaks</a>,<a href="http://arxiv.org/abs/2510.05262">ECPeaks</a>,<a href="https://doi.org/10.48550/arXiv.2510.25838">HPeaks</a>)</p>



<p class="wp-block-paragraph">We do have a promising candidate, though&#8212;IQP circuits! The fact that the output distributions of IQP circuits are quite simple could very well help us design sampling schemes with hidden secrets. Indeed, the idea of hiding peaks has been pioneered by Shepherd and Bremner (<a href="http://rspa.royalsocietypublishing.org/content/465/2105/1413">VerIQP1</a>) who found a way to design classically hard IQP circuits with a large hidden Fourier coefficient. The presence of this large Fourier coefficient can easily be checked from a few classical samples, and random IQP circuits do not have any large Fourier coefficients. Unfortunately, for that construction and a variation thereof (<a href="https://link.aps.org/doi/10.1103/PRXQuantum.6.020315">VerIQP2</a>), it turned out that the large coefficient can be detected quite easily from the circuit description (<a href="http://arxiv.org/abs/1912.05547">ClassIQP1</a>,<a href="http://arxiv.org/abs/2312.10156">ClassIQP2</a>).</p>



<p class="wp-block-paragraph">To this day, it remains an exciting open question whether secrets can be planted in (maybe IQP) circuit families in a way that allows for efficient classical verification. Even finding a scheme with some large gap between verification and simulation times would be exciting, because it would for the first time allow us to verify a quantum computing experiment in the advantage regime using only classical computation.</p>



<h4 class="wp-block-heading">Towards applications: certifiable random number generation</h4>



<p class="wp-block-paragraph">Beyond verified quantum advantage, sampling schemes with hidden secrets may be usable to generate classically certifiable random numbers: You sample from the output distribution of a random circuit with a planted secret, and verify that the samples come from the correct distribution using the secret. If the distribution has sufficiently high entropy, truly random numbers can be extracted from them. The same can be done for RCS, except that some acrobatics are needed to get around the problem that verification is just as costly as simulation (<a href="https://dl.acm.org/doi/10.1145/3564246.3585145">CertRand</a>, <a href="https://doi.org/10.48550/arXiv.2511.03686">CertRandExp</a>). Again, a large gap between verification and simulation times would probably permit such certified random number generation.</p>



<p class="wp-block-paragraph">The goal here is firstly a theoretical one: Come up with a planted-secret RCS scheme that has a large verification-simulation gap. But then, of course, it is an experimental one: actually perform such an experiment to classically verify quantum advantage.</p>



<p class="wp-block-paragraph">Should an IQP-based scheme of circuits with secrets exist, 100 logical qubits is the regime where it should give a relevant advantage.</p>



<h2 class="wp-block-heading">Three milestones</h2>



<p class="wp-block-paragraph">Altogether, I proposed three milestones for the 100 logical qubit regime.</p>



<ol class="wp-block-list">
<li>Perform fault-tolerant quantum advantage using random IQP circuits. This will allow an improvement of the fidelity towards performing near-perfect RCS and thus closes the scalability worries of noisy quantum advantage I discussed in my last post.</li>



<li>Perform RCS with symmetries. This will allow for efficient verification of quantum advantage in the trusted experimenter scenario and thus make a first step toward closing the verification loophole.</li>



<li>Find and perform RCS schemes with planted secrets. This will allow us to verify quantum advantage in the remote untrusted server scenario and presumably give a first useful application of quantum computers to generate classically certified random numbers.</li>
</ol>



<p class="wp-block-paragraph">All of these experiments are natural steps towards performing actually useful quantum algorithms in that they use more structured circuits than just random universal circuits and can be used to benchmark the performance of the quantum devices in an advantage regime. Moreover, all of them close some loophole of the previous quantum advantage demonstrations, just like follow-up experiments to the first Bell tests have closed the loopholes one by one.</p>



<p class="wp-block-paragraph">I argued that IQP circuits will play an important role in achieving those milestones since they are a natural circuit family in fault-tolerant constructions and promising candidates for random circuit constructions with planted secrets. Developing a better understanding of the properties of the output distributions of IQP circuits will help us achieve the theory challenges ahead.</p>



<p class="wp-block-paragraph">Experimentally, the 100 logical qubit regime is exactly the regime to shoot for with those circuits since while IQP circuits are somewhat easier to simulate than universal random circuits, 100 qubits is well in the classically intractable regime.</p>



<hr class="wp-block-separator has-alpha-channel-opacity" />



<h2 class="wp-block-heading">What I did not talk about</h2>



<p class="wp-block-paragraph">Let me close this mini-series by touching on a few things that I would have liked to discuss more.</p>



<p class="wp-block-paragraph">First, there is the OTOC experiment by the Google team (<a href="https://www.nature.com/articles/s41586-025-09526-6">OTOC</a>) which has spawned quite a debate. This experiment claims to achieve quantum advantage for an arguably more natural task than sampling, namely, computing expectation values. Computing expectation values is at the heart of quantum-chemistry and condensed-matter applications of quantum computers. And it has the nice property that it is what the Google team called &#8220;quantum-verifiable&#8221; (and what I would call &#8220;hopefully-in-the-future-verifiable&#8221;) in the following sense: Suppose we perform an experiment to measure a classically hard expectation value on a noisy device now, and suppose this expectation value actually carries some signal, so it is significantly far away from zero. Once we have a trustworthy quantum computer in the future, we will be able to check that the outcome of this experiment was correct and hence quantum advantage was achieved. There is a lot of interesting science to discuss about the details of this experiment and maybe I will do so in a future post.</p>



<p class="wp-block-paragraph">Finally, I want to mention an interesting theory challenge that relates to the noise-scaling arguments I discussed in detail in <a href="https://quantumfrontiers.com/2026/01/25/has-quantum-advantage-been-achieved-part-2-considering-the-evidence/">Part 2</a>: The challenge is to understand whether quantum advantage can be achieved in the presence of a constant amount of local noise. What do we know about this? On the one hand, log-depth random circuits with constant local noise are easy to simulate classically (<a href="https://quantum-journal.org/papers/q-2017-04-25-8/">SimIQP</a>,<a href="http://arxiv.org/abs/2211.03999">SimRCS</a>), and we have good numerical evidence that random circuits at very low depths are easy to simulate classically even without noise (<a href="https://link.aps.org/doi/10.1103/PhysRevX.12.021021">LowDSim</a>). So is there a depth regime in between the very low depth and the log-depth regime in which quantum advantage persists under constant local noise? Is this maybe even true in a noise regime that does not permit fault-tolerance (see <a href="https://www.youtube.com/watch?v=sH3XmKS-LeE">this interesting talk</a>)? In the regime in which fault-tolerance is possible, it turns out that <a href="https://www.youtube.com/watch?v=K7Z5M0MHUYQ">one can construct</a> simple fault-tolerance schemes that do not require any quantum feedback, so there are distributions that are hard to simulate classically even in the presence of constant local noise.</p>



<h2 class="wp-block-heading">So long, and thanks for all the fish!</h2>



<p class="wp-block-paragraph">I hope that in this mini-series I could convince you that quantum advantage has been achieved. There are some open loopholes but if you are happy with physics-level experimental evidence, then you should be convinced that the RCS experiments of the past years have demonstrated quantum advantage.</p>



<p class="wp-block-paragraph">As the devices are getting better at a rapid pace, there is a clear goal that I hope will be achieved in the 100-logical-qubit regime: demonstrate fault-tolerant and verifiable advantage (for the experimentalists) and come up with the schemes to do that (for the theorists)! Those experiments would close the loopholes of the current RCS experiments. And they would work as a stepping stone towards actual algorithms in the advantage regime.</p>



<hr class="wp-block-separator has-alpha-channel-opacity" />



<p class="wp-block-paragraph">I want to end with a huge thanks to Spiros Michalakis, John Preskill and Frederik Hahn who have patiently read and helped me improve these posts!</p>
